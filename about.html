𝚃𝚑𝚎𝚒𝚛 𝚚𝚞𝚎𝚜𝚝 𝚎𝚗𝚝𝚊𝚒𝚕𝚜 𝚗𝚘𝚝 𝚓𝚞𝚜𝚝 𝚊𝚗𝚗𝚒𝚑𝚒𝚕𝚊𝚝𝚒𝚗𝚐 𝚘𝚋𝚜𝚝𝚊𝚌𝚕𝚎𝚜, 𝚋𝚞𝚝 𝚊𝚕𝚜𝚘 𝚞𝚗𝚎𝚊𝚛𝚝𝚑𝚒𝚗𝚐 𝚎𝚡𝚝𝚛𝚎𝚖𝚎𝚕𝚢 𝚞𝚗𝚌𝚘𝚖𝚖𝚘𝚗 𝚊𝚗𝚍 𝚘𝚏𝚝𝚎𝚗 𝚑𝚊𝚣𝚊𝚛𝚍𝚘𝚞𝚜 𝚜𝚞𝚋𝚜𝚝𝚊𝚗𝚌𝚎𝚜 𝚏𝚛𝚘𝚖 𝚝𝚑𝚎 𝚍𝚞𝚗𝚐𝚎𝚘𝚗𝚜, 𝚖𝚒𝚡𝚒𝚗𝚐 𝚊 𝚑𝚞𝚗𝚝 𝚏𝚘𝚛 𝚝𝚛𝚎𝚊𝚜𝚞𝚛𝚎 𝚠𝚒𝚝𝚑 𝚊 𝚌𝚞𝚕𝚒𝚗𝚊𝚛𝚢 𝚓𝚘𝚞𝚛𝚗𝚎𝚢. 𝚃𝚑𝚒𝚜 𝚌𝚘𝚖𝚋𝚒𝚗𝚊𝚝𝚒𝚘𝚗 𝚘𝚏 𝚏𝚊𝚗𝚝𝚊𝚜𝚢, 𝚑𝚞𝚖𝚘𝚛, 𝚊𝚗𝚍 𝚏𝚘𝚘𝚍 𝚌𝚞𝚕𝚝𝚞𝚛𝚎 𝚙𝚛𝚘𝚟𝚒𝚍𝚎𝚜 𝚊 𝚘𝚗𝚎-𝚘𝚏-𝚊-𝚔𝚒𝚗𝚍 𝚊𝚗𝚍 𝚎𝚗𝚝𝚎𝚛𝚝𝚊𝚒𝚗𝚒𝚗𝚐 𝚎𝚡𝚙𝚎𝚛𝚒𝚎𝚗𝚌𝚎 𝚏𝚘𝚛 𝚊𝚏𝚒𝚌𝚒𝚘𝚗𝚊𝚍𝚘𝚜 𝚘𝚏 𝚋𝚘𝚝𝚑 𝚐𝚎𝚗𝚛𝚎𝚜.

𝕎𝕙𝕪 𝔻𝕦𝕟𝕘𝕖𝕠𝕟 𝕄𝕖𝕤𝕙𝕚?

𝙸 𝚏𝚘𝚞𝚗𝚍 𝚝𝚑𝚎 𝚌𝚘𝚗𝚌𝚎𝚙𝚝 𝚝𝚘 𝚑𝚊𝚟𝚎 𝚙𝚘𝚝𝚎𝚗𝚝𝚒𝚊𝚕 𝚠𝚒𝚝𝚑 𝚒𝚝𝚜 𝚌𝚛𝚎𝚊𝚝𝚒𝚟𝚒𝚝𝚢 𝚒𝚗 𝚊 𝚠𝚘𝚛𝚕𝚍 𝚠𝚑𝚎𝚛𝚎 𝚝𝚑𝚎𝚛𝚎 𝚊𝚛𝚎 𝚖𝚊𝚐𝚎𝚜, 𝚎𝚕𝚟𝚎𝚜, & 𝚔𝚗𝚒𝚐𝚑𝚝𝚜 𝚐𝚘𝚒𝚗𝚐 𝚝𝚑𝚛𝚘𝚞𝚐𝚑 𝚝𝚑𝚎 𝚝𝚛𝚒𝚊𝚕𝚜 𝚘𝚏 𝚝𝚑𝚎 𝚍𝚞𝚗𝚐𝚎𝚘𝚗, 𝚒𝚗 𝚘𝚛𝚍𝚎𝚛 𝚝𝚘 𝚊𝚌𝚑𝚒𝚎𝚟𝚎 𝚝𝚑𝚎 𝚞𝚗𝚏𝚘𝚞𝚗𝚍 "𝚝𝚛𝚎𝚊𝚜𝚞𝚛𝚎". 𝚈𝚎𝚝 𝚢𝚘𝚞'𝚍 𝚑𝚊𝚟𝚎 𝚝𝚘 𝚋𝚎 𝚛𝚎𝚜𝚘𝚞𝚛𝚌𝚎𝚏𝚞𝚕 𝚠𝚒𝚝𝚑 𝚢𝚘𝚞𝚛 𝚒𝚗𝚜𝚝𝚒𝚗𝚌𝚝𝚜, 𝚘𝚗𝚕𝚢 𝚝𝚘 𝚋𝚎 𝚜𝚞𝚛𝚙𝚊𝚜𝚜𝚒𝚗𝚐 𝚌𝚛𝚎𝚊𝚝𝚞𝚛𝚎𝚜 𝚝𝚑𝚛𝚘𝚞𝚐𝚑𝚘𝚞𝚝 𝚝𝚑𝚎 𝚍𝚞𝚗𝚐𝚎𝚘𝚗-- 𝚊𝚗𝚍 𝚖𝚘𝚜𝚝 𝚒𝚖𝚙𝚘𝚛𝚝𝚊𝚗𝚝𝚕𝚢, 𝘵𝘰 𝘦𝘢𝘵 𝘵𝘩𝘦𝘮.
